Пример реализации перегрузки методов


// Пример перегрузки метода applyDamage/heal
applyDamage(value: number): void;
applyDamage(action: 'heal' | 'damage', value: number): void;
applyDamage(param1: number | 'heal' | 'damage', param2?: number): void {
    // Реализация метода с проверкой типа параметров
    if (typeof param1 === 'number') {
        // Первая форма: applyDamage(value: number)
        const amount = param1;
        this.health = Math.max(0, Math.min(this.maxHealth, this.health + amount));
        this.addLog(amount >= 0 ? `Получено лечение: +${amount} HP` : 
                               `Получен урон: ${amount} HP`);
    } else {
        // Вторая форма: applyDamage(action, value)
        const action = param1;
        const amount = param2 || 0;
        const actualAmount = action === 'heal' ? amount : -amount;
        this.health = Math.max(0, Math.min(this.maxHealth, this.health + actualAmount));
        this.addLog(`${action === 'heal' ? 'Лечение' : 'Урон'}: ${amount} HP`);
    }
}
